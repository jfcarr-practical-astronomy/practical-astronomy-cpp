530

557  double l_1_deg = 279.6967 + (0.000303 * t_centuries * t_centuries);
558  double l_deg1 = l_1_deg + 360 * (a_deg - floor(a_deg));
559  double l_deg2 = l_deg1 - 360 * floor(l_deg1 / 360);
560  double l_rad = degrees_to_radians(l_deg2);
561  double b_deg = 5.372617 * t_centuries;
562  double n_deg1 = 259.1833 - 360 * (b_deg - floor(b_deg));
563  double n_deg2 = n_deg1 - 360 * (floor(n_deg1 / 360));
564  double n_rad = degrees_to_radians(n_deg2);
565  double nut_in_long_arcsec = -17.2 * sin(n_rad) - 1.3 * sin(2 * l_rad);
566  double nut_in_obl_arcsec = 9.2 * cos(n_rad) + 0.5 * cos(2 * l_rad);

569  double nut_in_obl_deg = nut_in_obl_arcsec / 3600;
